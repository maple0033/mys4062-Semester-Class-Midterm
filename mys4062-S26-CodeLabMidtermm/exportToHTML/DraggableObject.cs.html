using System;
using System.IO;
using UnityEngine;

public class DraggableObject : MonoBehaviour
{
    private Vector3 currentMousePosition;
    
    string  filePath = "/Resources/DraggableObject/";
    // Start is called once before the first execution of Update after the MonoBehaviour is created
    void Start()
    {
        filePath = Application.dataPath + filePath; //set filepath 
        string fileContents = File.ReadAllText(filePath + name + ".json"); //lets put the saved data in a string
        Debug.Log(fileContents); //print it out
        
        Vector3 savePosition = JsonUtility.FromJson<Vector3>(fileContents); //turn it back into a vector 3 for us
        transform.position = savePosition; //set position to that
    }

    // Update is called once per frame
    void Update()
    {
        
    }

    private void OnMouseDrag()
    {
        transform.position = GetMousePosition();
    }

    Vector3 GetMousePosition() //this function lets you check if the mouse drags 
    {
        currentMousePosition =  Input.mousePosition;
        currentMousePosition.z = Camera.main.WorldToScreenPoint(transform.position).z;
        //clipping plane near = is something is near this, it will render
        //clipping plane far = if something is far, unnecessary things will not get rendered
        
        //now the mouse is a current point on your screen, the game can track that
        //it also knows how DEEP it is because of the transform.position

        currentMousePosition = Camera.main.ScreenToWorldPoint(currentMousePosition);
        //this is gonna take the position on the screen and put it back to the world

        return currentMousePosition; //we will return this value as a VECTOR3 as indicated by the 
        //function being a Vector3 not a void

    }

    void OnApplicationQuit()
    {
        string jsonPosition = JsonUtility.ToJson(transform.position, true); //if you say false it will not pretty print
        Debug.Log(jsonPosition);
        
        File.WriteAllText(filePath + name + ".json", jsonPosition); //
    }
}
